def _on_performance_center(self):
    """打开FactorWeave智能性能监控中心"""
    try:
        from gui.widgets.modern_performance_widget import show_modern_performance_monitor
        from core.performance.factorweave_performance_integration import get_performance_integrator
        
        # 获取智能监控数据
        integrator = get_performance_integrator()
        dashboard_data = integrator.get_performance_dashboard_data()
        
        # 显示现代化性能监控界面
        performance_widget = show_modern_performance_monitor(self._main_window)
        performance_widget.setWindowTitle("FactorWeave-Quant 智能性能监控中心 - Professional Edition")
        
        # 如果有智能洞察，在状态栏显示提示
        insights_count = len(dashboard_data.get('insights', []))
        if insights_count > 0:
            performance_widget.status_bar.showMessage(f"发现 {insights_count} 条智能性能洞察")
        
        performance_widget.show()
        logger.info("FactorWeave智能性能监控中心已打开")
